Database Schema

admissions system — 22 tables across 6 domains

The admissions database is built on PostgreSQL with 22 tables organized into six logical groups: identity, application form, events, analysis, leads, and onboarding. See schema.sql in this folder for the complete DDL.

    applicant {
        serial applicant_id PK
        varchar email UK
        varchar first_name
        varchar last_name
        varchar password_hash
        boolean verified
        varchar referral_source
    }

    applicant_stage {
        uuid stage_id PK
        integer applicant_id FK
        varchar current_stage
        varchar program_admission_status
        boolean pledge_completed
        boolean deferred
        varchar deliberation
        uuid current_cohort_id FK
    }

    application {
        serial application_id PK
        integer applicant_id FK
        varchar status
        timestamp submitted_at
    }

    section {
        integer section_id PK
        varchar name
        integer display_order
    }

    question {
        integer question_id PK
        integer section_id FK
        text prompt
        varchar response_type
        boolean is_required
        boolean active
        integer parent_question_id FK
    }

    response {
        serial response_id PK
        integer application_id FK
        integer question_id FK
        text response_value
    }

    application_analysis {
        serial analysis_id PK
        integer application_id FK
        integer overall_score
        varchar recommendation
        varchar final_status
        boolean has_masters_degree
    }

    event {
        uuid event_id PK
        uuid type_id FK
        varchar title
        timestamp start_time
        integer capacity
        boolean is_online
        varchar workshop_type
    }

    event_registration {
        serial registration_id PK
        uuid event_id FK
        integer applicant_id FK
        varchar status
        boolean needs_laptop
    }

    lead {
        serial lead_id PK
        varchar email UK
        varchar status
        integer applicant_id FK
    }

    lead_source {
        serial source_id PK
        integer lead_id FK
        varchar source_type
        varchar source_name
    }

    lead_engagement {
        serial engagement_id PK
        integer lead_id FK
        varchar engagement_type
    }

    onboarding_tasks {
        serial task_id PK
        varchar title
        boolean is_required
        integer display_order
    }
